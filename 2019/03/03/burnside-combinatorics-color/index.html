  ll ret = 1;

inline ll C(int N, int M) {
  ll ret = 1;
  for (int i = 2; i <= M; ++i) ret = (ret * i) % mod;
  ret = quick_power(ret, mod - 2);
  for (int i = N; i > N - M; --i)
    ret = (ret * i) % mod;
  return ret;
}
